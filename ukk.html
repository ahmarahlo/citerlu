<?php
session_start();
include 'koneksi.php';
$proses = isset($_GET['proses']) ? $_GET['proses'] : '';

if ($proses == 'register') {
    $username = $_POST['username']; $password = $_POST['password'];
    mysqli_query($koneksi, "INSERT INTO users (username, password, role) VALUES ('$username', '$password', 'Pelanggan')");
    header("Location: login.php?msg=register_sukses");
} elseif ($proses == 'tambah_laptop') {
    mysqli_query($koneksi, "INSERT INTO laptop (nama_laptop, jenis, id_pemilik) VALUES ('$_POST[nama_laptop]', '$_POST[jenis]', '$_SESSION[id]')");
    header("Location: form_laptop.php");
} elseif ($proses == 'edit_laptop') {
    mysqli_query($koneksi, "UPDATE laptop SET nama_laptop='$_POST[nama_laptop]', jenis='$_POST[jenis]' WHERE id='$_POST[id]' AND id_pemilik='$_SESSION[id]'");
    header("Location: form_laptop.php");
} elseif ($proses == 'hapus_laptop') {
    mysqli_query($koneksi, "DELETE FROM laptop WHERE id='$_GET[id]'");
    header("Location: form_laptop.php");
} elseif ($proses == 'tambah_layanan') {
    mysqli_query($koneksi, "INSERT INTO servis (nama_servis, harga_servis) VALUES ('$_POST[nama_servis]', '$_POST[harga_servis]')");
    header("Location: form_servis.php");
} elseif ($proses == 'edit_layanan') {
    mysqli_query($koneksi, "UPDATE servis SET nama_servis='$_POST[nama_servis]', harga_servis='$_POST[harga_servis]' WHERE id='$_POST[id]'");
    header("Location: form_servis.php");
} elseif ($proses == 'update_status') {
    $id = $_POST['id_booking']; $st = $_POST['status'];
    mysqli_query($koneksi, "UPDATE booking SET status='$st' WHERE id='$id'");
    if($st == 'Selesai') {
        $q_jasa = mysqli_fetch_assoc(mysqli_query($koneksi, "SELECT s.harga_servis FROM booking b JOIN servis s ON b.id_servis = s.id WHERE b.id='$id'"));
        $q_p = mysqli_fetch_assoc(mysqli_query($koneksi, "SELECT SUM(qty*harga_satuan) as tp FROM detail_servis WHERE id_booking='$id'"));
        $total = $q_jasa['harga_servis'] + $q_p['tp'];
        $tgl_s = date('Y-m-d'); $g_s = date('Y-m-d', strtotime('+30 days'));
        mysqli_query($koneksi, "INSERT INTO nota_pembayaran (id_booking, total_biaya, tanggal_selesai, garansi_servis_habis) VALUES ('$id', '$total', '$tgl_s', '$g_s')");
    }
    header("Location: dashboard.php");
} elseif ($proses == 'tambah_sparepart_ke_servis') {
    $id_b = $_POST['id_booking']; $id_s = $_POST['id_sparepart']; $qty = $_POST['qty'];
    $stok = mysqli_fetch_assoc(mysqli_query($koneksi, "SELECT jumlah, harga_jual FROM sparepart WHERE id='$id_s'"));
    if($stok['jumlah'] >= $qty) {
        $h = $stok['harga_jual'];
        mysqli_query($koneksi, "UPDATE sparepart SET jumlah = jumlah - $qty WHERE id='$id_s'");
        mysqli_query($koneksi, "INSERT INTO detail_servis (id_booking, id_sparepart, qty, harga_satuan) VALUES ('$id_b', '$id_s', '$qty', '$h')");
    }
    header("Location: dashboard.php");
}
?>
